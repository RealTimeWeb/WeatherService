001    package realtimeweb.weatherservice.json;
002    
003    import realtimeweb.weatherservice.main.AbstractWeatherService;
004    import java.util.HashMap;
005    import realtimeweb.weatherservice.util.Util;
006    
007    /**
008     * Used to get data as a raw string.
009     */
010    public class JsonWeatherService implements AbstractWeatherService {
011            private static JsonWeatherService instance;
012            protected boolean local;
013            private ClientStore clientStore;
014            /**
015             * **For internal use only!** Protected Constructor guards against instantiation.
016            
017             * @return 
018             */
019            protected  JsonWeatherService() {
020                    disconnect();
021                    this.clientStore = new ClientStore();
022            }
023            
024            /**
025             * Retrieves the singleton instance.
026            
027             * @return JsonWeatherService
028             */
029            public static JsonWeatherService getInstance() {
030                    if (instance == null) {
031                            synchronized (JsonWeatherService.class) {
032                                    if (instance == null) {
033                                            instance = new JsonWeatherService();
034                                    }
035                            }
036                            
037                    }
038                    return instance;
039            }
040            
041            /**
042             * Establishes a connection to the online service. Requires an internet connection.
043            
044             */
045            @Override
046            public void connect() {
047                    this.local = false;
048            }
049            
050            /**
051             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
052            
053             */
054            @Override
055            public void disconnect() {
056                    this.local = true;
057            }
058            
059            /**
060             * **For internal use only!** The ClientStore is the internal cache where offline data is stored.
061            
062             * @return ClientStore
063             */
064            public ClientStore getClientStore() {
065                    return this.clientStore;
066            }
067            
068            /**
069             * Gets a report on the current weather, forecast, and more detailed information about the location.
070             * @param latitude The latitude (up-down) of the location to get information about.
071             * @param longitude The longitude (left-right) of the location to get information about.
072             * @return String
073             */
074            public String getReport(double latitude, double longitude) throws Exception {
075                    String url = "http://forecast.weather.gov/MapClick.php";
076                    HashMap<String, String> parameters = new HashMap<String, String>();
077                    parameters.put("lat", String.valueOf(latitude));
078                    parameters.put("lon", String.valueOf(longitude));
079                    if (this.local) {
080                            return clientStore.getData(Util.hashRequest(url, parameters));
081                    }
082                    parameters.put("FcstType", String.valueOf("json"));
083                    String jsonResponse = "";
084                    try {
085                        jsonResponse = Util.get(url, parameters);
086                        if (jsonResponse.startsWith("<")) {
087                            throw new Exception(jsonResponse);
088                        }
089                        return jsonResponse;
090                    } catch (Exception e) {
091                        throw new Exception(e.toString());
092                    }
093            }
094            
095            /**
096             * Gets a report on the current weather, forecast, and more detailed information about the location.
097             * @param latitude The latitude (up-down) of the location to get information about.
098             * @param longitude The longitude (left-right) of the location to get information about.
099             * @param callback The listener that will be given the data (or error).
100             */
101            public void getReport(final double latitude, final double longitude, final JsonGetReportListener callback) {
102                    
103                    Thread thread = new Thread() {
104                        @Override
105                        public void run() {
106                            try {
107                                callback.getReportCompleted(JsonWeatherService.getInstance().getReport(latitude, longitude));
108                            } catch (Exception e) {
109                                callback.getReportFailed(e);
110                            }
111                        }
112                    };
113                    thread.start();
114                    
115            }
116            
117    }
