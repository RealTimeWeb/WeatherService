010    
011    import java.util.ArrayList;
012    import java.util.Arrays;

015    
016    /**
017     * A detailed description of a location
018     */
019    public class Location {
020            
021            
022            private double latitude;
023            private double longitude;
024            private int elavation;
025            private String name;
026            
027            
028            /**
029             * The latitude (up-down) of this location.
030            
031             * @return double
032             */
033            public double getLatitude() {
034                    return this.latitude;
035            }
036            
037            /**
038             * 
039             * @param latitude The latitude (up-down) of this location.
040             */
041            public void setLatitude(double latitude) {
042                    this.latitude = latitude;
043            }
044            
045            /**
046             * The longitude (left-right) of this location.
047            
048             * @return double
049             */
050            public double getLongitude() {
051                    return this.longitude;
052            }
053            
054            /**
055             * 
056             * @param longitude The longitude (left-right) of this location.
057             */
058            public void setLongitude(double longitude) {
059                    this.longitude = longitude;
060            }
061            
062            /**
063             * The height above sea-level (in feet).
064            
065             * @return int
066             */
067            public int getElavation() {
068                    return this.elavation;
069            }
070            
071            /**
072             * 
073             * @param elavation The height above sea-level (in feet).
074             */
075            public void setElavation(int elavation) {
076                    this.elavation = elavation;
077            }
078            
079            /**
080             * The city and state that this location is in.
081            
082             * @return String
083             */
084            public String getName() {
085                    return this.name;
086            }
087            
088            /**
089             * 
090             * @param name The city and state that this location is in.
091             */
092            public void setName(String name) {
093                    this.name = name;
094            }
095            
096            
097            
098            /**
099             * A detailed description of a location
100            
101             * @return String
102             */
103            public String toString() {
104                    return "Location[" + latitude + ", " + longitude + ", " + elavation + ", " + name + "]";
105            }
106            
107            /**
108             * Internal constructor to create a Location from a Json representation.
109             * @param json The raw json data that will be parsed.
110             * @param gson The Gson parser. See <a href='https://code.google.com/p/google-gson/'>https://code.google.com/p/google-gson/</a> for more information.
111             * @return 
112             */
113            public  Location(JsonObject json, Gson gson) {
114                    this.latitude = json.get("latitude").getAsDouble();
115                    this.longitude = json.get("longitude").getAsDouble();
116                    this.elavation = json.get("elevation").getAsInt();
117                    this.name = json.get("areaDescription").getAsString();
118            }
119            
120            /**
121             * Regular constructor to create a Location.
122             * @param latitude The latitude (up-down) of this location.
123             * @param longitude The longitude (left-right) of this location.
124             * @param elavation The height above sea-level (in feet).
125             * @param name The city and state that this location is in.
126             * @return 
127             */
128            public  Location(double latitude, double longitude, int elavation, String name) {
129                    this.latitude = latitude;
130                    this.longitude = longitude;
131                    this.elavation = elavation;
132                    this.name = name;
133            }
134            
135    }
