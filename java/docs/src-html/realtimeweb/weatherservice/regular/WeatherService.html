001    package realtimeweb.weatherservice.regular;
002    
003    import realtimeweb.weatherservice.main.AbstractWeatherService;
004    import realtimeweb.weatherservice.json.JsonWeatherService;
005    import realtimeweb.weatherservice.util.Util;
006    import java.util.ArrayList;
007    import java.util.HashMap;
008    import java.util.Map;
009    import com.google.gson.Gson;
010    import com.google.gson.GsonBuilder;
011    import com.google.gson.JsonArray;
012    import com.google.gson.JsonObject;
013    import com.google.gson.JsonParser;
014    import realtimeweb.weatherservice.domain.Report;
015    import realtimeweb.weatherservice.json.JsonGetReportListener;
016    
017    /**
018     * Used to get data as classes.
019     */
020    public class WeatherService implements AbstractWeatherService {
021            private static WeatherService instance;
022            private JsonWeatherService jsonInstance;
023            private Gson gson;
024            /**
025             * **For internal use only!** Protected Constructor guards against instantiation.
026            
027             * @return 
028             */
029            protected  WeatherService() {
030                    this.jsonInstance = JsonWeatherService.getInstance();
031                    this.gson = new Gson();
032            }
033            
034            /**
035             * Retrieves the singleton instance.
036            
037             * @return WeatherService
038             */
039            public static WeatherService getInstance() {
040                    if (instance == null) {
041                            synchronized (WeatherService.class) {
042                                    if (instance == null) {
043                                            instance = new WeatherService();
044                                    }
045                            }
046                            
047                    }
048                    return instance;
049            }
050            
051            /**
052             * Establishes a connection to the online service. Requires an internet connection.
053            
054             */
055            @Override
056            public void connect() {
057                    jsonInstance.connect();
058            }
059            
060            /**
061             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
062            
063             */
064            @Override
065            public void disconnect() {
066                    jsonInstance.disconnect();
067            }
068            
069            /**
070             * Gets a report on the current weather, forecast, and more detailed information about the location.
071             * @param latitude The latitude (up-down) of the location to get information about.
072             * @param longitude The longitude (left-right) of the location to get information about.
073             * @return Report
074             */
075            public Report getReport(double latitude, double longitude) throws Exception {
076                    String response = jsonInstance.getReport(latitude,longitude);
077                    JsonParser parser = new JsonParser();
078                    JsonObject top = parser.parse(response).getAsJsonObject();
079                    return new Report(top, gson);
080            }
081            
082            /**
083             * Gets a report on the current weather, forecast, and more detailed information about the location.
084             * @param latitude The latitude (up-down) of the location to get information about.
085             * @param longitude The longitude (left-right) of the location to get information about.
086             * @param callback The listener that will receive the data (or error).
087             */
088            public void getReport(double latitude, double longitude, final GetReportListener callback) {
089                    
090                    jsonInstance.getReport(latitude, longitude, new JsonGetReportListener() {
091                        @Override
092                        public void getReportFailed(Exception exception) {
093                            callback.getReportFailed(exception);
094                        }
095                        
096                        @Override
097                        public void getReportCompleted(String response) {
098                            JsonParser parser = new JsonParser();
099                    JsonObject top = parser.parse(response).getAsJsonObject();
100                            Report result = new Report(top, gson);
101                            callback.getReportCompleted(result);
102                        }
103                    });
104                    
105            }
106            
107    }
