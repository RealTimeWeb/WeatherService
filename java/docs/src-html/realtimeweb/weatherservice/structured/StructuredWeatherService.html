001    package realtimeweb.weatherservice.structured;
002    
003    import realtimeweb.weatherservice.main.AbstractWeatherService;
004    import realtimeweb.weatherservice.util.Util;
005    import java.util.HashMap;
006    import java.util.LinkedHashMap;
007    import com.google.gson.Gson;
008    import realtimeweb.weatherservice.json.JsonWeatherService;
009    import realtimeweb.weatherservice.json.JsonGetReportListener;
010    
011    /**
012     * Used to get data as built-in Java objects (HashMap, ArrayList, etc.).
013     */
014    public class StructuredWeatherService implements AbstractWeatherService {
015            private static StructuredWeatherService instance;
016            private JsonWeatherService jsonInstance;
017            private Gson gson;
018            /**
019             * **For internal use only!** Protected Constructor guards against instantiation.
020            
021             * @return 
022             */
023            protected  StructuredWeatherService() {
024                    this.jsonInstance = JsonWeatherService.getInstance();
025                    this.gson = new Gson();
026            }
027            
028            /**
029             * Retrieves the singleton instance.
030            
031             * @return StructuredWeatherService
032             */
033            public static StructuredWeatherService getInstance() {
034                    if (instance == null) {
035                            synchronized (StructuredWeatherService.class) {
036                                    if (instance == null) {
037                                            instance = new StructuredWeatherService();
038                                    }
039                            }
040                            
041                    }
042                    return instance;
043            }
044            
045            /**
046             * Establishes a connection to the online service. Requires an internet connection.
047            
048             */
049            @Override
050            public void connect() {
051                    jsonInstance.connect();
052            }
053            
054            /**
055             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
056            
057             */
058            @Override
059            public void disconnect() {
060                    jsonInstance.disconnect();
061            }
062            
063            /**
064             * Gets a report on the current weather, forecast, and more detailed information about the location.
065             * @param latitude The latitude (up-down) of the location to get information about.
066             * @param longitude The longitude (left-right) of the location to get information about.
067             * @return HashMap<String, Object>
068             */
069            public HashMap<String, Object> getReport(double latitude, double longitude) throws Exception {
070                    return gson.fromJson(jsonInstance.getReport(latitude, longitude), LinkedHashMap.class);
071            }
072            
073            /**
074             * Gets a report on the current weather, forecast, and more detailed information about the location.
075             * @param latitude The latitude (up-down) of the location to get information about.
076             * @param longitude The longitude (left-right) of the location to get information about.
077             * @param callback The listener that will be given the data (or error)
078             */
079            public void getReport(double latitude, double longitude, final StructuredGetReportListener callback) {
080                    
081                    jsonInstance.getReport(latitude, longitude, new JsonGetReportListener() {
082                        @Override
083                        public void getReportFailed(Exception exception) {
084                            callback.getReportFailed(exception);
085                        }
086                        
087                        @Override
088                        public void getReportCompleted(String data) {
089                            callback.getReportCompleted(gson.fromJson(data, LinkedHashMap.class));
090                        }
091                    });
092                    
093            }
094            
095    }
